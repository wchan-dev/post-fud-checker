
-- Procedure to insert data into Comments
CREATE OR REPLACE PROCEDURE insert_into_comments(post_id INT, text TEXT, timestamp TIMESTAMP WITH TIME ZONE)
LANGUAGE plpgsql
AS $$
BEGIN
    INSERT INTO Comments(Post_ID, Text, Timestamp) VALUES (post_id, text, timestamp);
END; $$

-- Procedure to insert data into Sentiments
CREATE OR REPLACE PROCEDURE insert_into_sentiments(comment_id INT, compound REAL, neg REAL, neu REAL, pos REAL)
LANGUAGE plpgsql
AS $$
BEGIN
    INSERT INTO Sentiments(Comment_ID, Compound, Neg, Neu, Pos) VALUES (comment_id, compound, neg, neu, pos);
END; $$

-- Procedure to retrieve comments for a particular post
CREATE OR REPLACE PROCEDURE get_comments_for_post(post_id INT)
LANGUAGE plpgsql
AS $$
BEGIN
    SELECT * FROM Comments WHERE Post_ID = post_id;
END; $$

-- Procedure to retrieve sentiment for a particular comment
CREATE OR REPLACE PROCEDURE get_sentiment_for_comment(comment_id INT)
LANGUAGE plpgsql
AS $$
BEGIN
    SELECT * FROM Sentiments WHERE Comment_ID = comment_id;
END; $$
